1. Given a sorted array and a number X, search two elements of the array such that their sum is X. Expected time complexity is O(n).
================================================================================

--- C Implementation ---
#include <stdio.h>

void findTwoSum(int arr[], int n, int X) {
    int left = 0, right = n - 1;
    while (left < right) {
        int sum = arr[left] + arr[right];
        if (sum == X) {
            printf("Elements found: %d and %d\n", arr[left], arr[right]);
            return;
        } else if (sum < X) {
            left++;
        } else {
            right--;
        }
    }
    printf("No two elements sum up to %d\n", X);
}

int main() {
    int arr[] = {1, 2, 4, 5, 7, 11};
    int n = sizeof(arr) / sizeof(arr[0]);
    int X = 9;
    printf("Array: ");
    for(int i=0; i<n; i++) printf("%d ", arr[i]);
    printf("\nTarget sum: %d\n", X);
    findTwoSum(arr, n, X);
    return 0;
}


--- Python Implementation ---
def find_two_sum(arr, x):
    left, right = 0, len(arr) - 1
    while left < right:
        current_sum = arr[left] + arr[right]
        if current_sum == x:

            return
        elif current_sum < x:
            left += 1
        else:
            right -= 1

--- C Implementation ---
#include <stdio.h>

int binarySearch(int arr[], int left, int right, int x) {
    if (right >= left) {
        int mid = left + (right - left) / 2;
        if (arr[mid] == x)
            return mid;
        if (arr[mid] > x)
            return binarySearch(arr, left, mid - 1, x);
        return binarySearch(arr, mid + 1, right, x);
    }
    return -1;
}

int main() {

        printf("Element is not present in array\n");
    else
        printf("Element is present at index %d\n", result);
    return 0;
}

void search2D(int r, int c, int matrix[r][c], int target) {
    int left = 0, right = r * c - 1;
    while (left <= right) {
        int mid = left + (right - left) / 2;
        int mid_val = matrix[mid / c][mid % c];
        if (mid_val == target) {
            printf("Found at row %d, col %d\n", mid / c, mid % c);
            return;
        } else if (mid_val < target) {
            left = mid + 1;
        } else {
            right = mid - 1;
        }
    }
    printf("Element not found.\n");
}

int mergeAndCount(int arr[], int temp[], int left, int mid, int right) {

            temp[k++] = arr[j++];

6. Implement a greedy algorithm to solve the fractional knapsack problem.
================================================================================

--- C Implementation ---
#include <stdio.h>
#include <stdlib.h>

struct Item {
    int value, weight;
};

int cmp(const void *a, const void *b) {
    struct Item *item1 = (struct Item *)a;
    struct Item *item2 = (struct Item *)b;
    double r1 = (double)item1->value / item1->weight;
    double r2 = (double)item2->value / item2->weight;
    if (r1 < r2) return 1;
    else if (r1 > r2) return -1;
    return 0;
}

double fractionalKnapsack(int W, struct Item arr[], int n) {
    qsort(arr, n, sizeof(arr[0]), cmp);
    int currentWeight = 0;
    double finalValue = 0.0;
    for (int i = 0; i < n; i++) {
        if (currentWeight + arr[i].weight <= W) {
            currentWeight += arr[i].weight;
            finalValue += arr[i].value;
        } else {
            int remain = W - currentWeight;
            finalValue += arr[i].value * ((double)remain / arr[i].weight);
            break;
        }
    }
    return finalValue;
}

int main() {
    int W = 50;
    struct Item arr[] = {{60, 10}, {100, 20}, {120, 30}};
    int n = sizeof(arr) / sizeof(arr[0]);
    printf("Maximum value we can obtain = %f\n", fractionalKnapsack(W, arr, n));
    return 0;
}


--- Python Implementation ---
class Item:
    def __init__(self, value, weight):
        self.value = value
        self.weight = weight

def fractional_knapsack(W, arr):
    arr.sort(key=lambda x: (x.value/x.weight), reverse=True)
    final_value = 0.0
    
    for item in arr:
        if W >= item.weight:
            W -= item.weight
            final_value += item.value
        else:
            final_value += item.value * (W / item.weight)
            break
            
    return final_value

if __name__ == "__main__":
    W = 50
    arr = [Item(60, 10), Item(100, 20), Item(120, 30)]
    max_val = fractional_knapsack(W, arr)
    print(f"Maximum value we can obtain = {max_val}")

7. Find the second largest and second smallest number simultaneously in an array using Divide & Conquer Principle.
================================================================================

--- C Implementation ---
#include <stdio.h>
#include <limits.h>

struct Result {
    int min, max, second_min, second_max;
};

struct Result findSecondMinMax(int arr[], int left, int right) {
    struct Result res, left_res, right_res;
    if (left == right) {
        res.min = arr[left];
        res.max = arr[left];
        res.second_min = INT_MAX;
        res.second_max = INT_MIN;
        return res;
    }
    if (right == left + 1) {
        if (arr[left] < arr[right]) {
            res.min = arr[left];
            res.second_min = arr[right];
            res.max = arr[right];
            res.second_max = arr[left];
        } else {
            res.min = arr[right];
            res.second_min = arr[left];
            res.max = arr[left];
            res.second_max = arr[right];
        }
        return res;
    }
    int mid = (left + right) / 2;
    left_res = findSecondMinMax(arr, left, mid);
    right_res = findSecondMinMax(arr, mid + 1, right);
    
    // Merge min and max
    if (left_res.min < right_res.min) {
        res.min = left_res.min;
        res.second_min = (left_res.second_min < right_res.min) ? left_res.second_min : right_res.min;
    } else {
        res.min = right_res.min;
        res.second_min = (right_res.second_min < left_res.min) ? right_res.second_min : left_res.min;
    }

    if (left_res.max > right_res.max) {
        res.max = left_res.max;
        res.second_max = (left_res.second_max > right_res.max) ? left_res.second_max : right_res.max;
    } else {
        res.max = right_res.max;
        res.second_max = (right_res.second_max > left_res.max) ? right_res.second_max : left_res.max;
    }
    return res;
}

int main() {
    int arr[] = {12, 11, 35, 10, 34, 1};
    int n = sizeof(arr) / sizeof(arr[0]);
    struct Result res = findSecondMinMax(arr, 0, n - 1);
    printf("Second Minimum: %d\n", res.second_min);
    printf("Second Maximum: %d\n", res.second_max);
    return 0;
}


--- Python Implementation ---
import sys

class Result:
    def __init__(self, min_val, max_val, sec_min, sec_max):
        self.min = min_val
        self.max = max_val
        self.second_min = sec_min
        self.second_max = sec_max

def find_second_min_max(arr, left, right):
    if left == right:
        return Result(arr[left], arr[left], sys.maxsize, -sys.maxsize)
    
    if right == left + 1:
        if arr[left] < arr[right]:
            return Result(arr[left], arr[right], arr[right], arr[left])
        else:
            return Result(arr[right], arr[left], arr[left], arr[right])
            
    mid = (left + right) // 2
    left_res = find_second_min_max(arr, left, mid)
    right_res = find_second_min_max(arr, mid + 1, right)
    
    # merge logic
    new_min = min(left_res.min, right_res.min)
    new_sec_min = min(max(left_res.min, right_res.min), left_res.second_min, right_res.second_min)
    
    new_max = max(left_res.max, right_res.max)
    new_sec_max = max(min(left_res.max, right_res.max), left_res.second_max, right_res.second_max)
    
    return Result(new_min, new_max, new_sec_min, new_sec_max)

if __name__ == "__main__":
    arr = [12, 11, 35, 10, 34, 1]
    res = find_second_min_max(arr, 0, len(arr) - 1)
    print(f"Second Minimum: {res.second_min}")
    print(f"Second Maximum: {res.second_max}")

8. Given a sorted array and a number x, write a function that counts the occurrences of x in the array. Expected time complexity is O(logn).
================================================================================

--- C Implementation ---
#include <stdio.h>

int binarySearchFirst(int arr[], int left, int right, int x) {
    if (right >= left) {
        int mid = (left + right) / 2;
        if ((mid == 0 || x > arr[mid - 1]) && arr[mid] == x) return mid;
        else if (x > arr[mid]) return binarySearchFirst(arr, mid + 1, right, x);
        else return binarySearchFirst(arr, left, mid - 1, x);
    }
    return -1;
}

int binarySearchLast(int arr[], int left, int right, int x, int n) {
    if (right >= left) {
        int mid = (left + right) / 2;
        if ((mid == n - 1 || x < arr[mid + 1]) && arr[mid] == x) return mid;
        else if (x < arr[mid]) return binarySearchLast(arr, left, mid - 1, x, n);
        else return binarySearchLast(arr, mid + 1, right, x, n);
    }
    return -1;
}

int countOccurrences(int arr[], int n, int x) {
    int first = binarySearchFirst(arr, 0, n - 1, x);
    if (first == -1) return 0;
    int last = binarySearchLast(arr, first, n - 1, x, n);
    return last - first + 1;
}

int main() {
    int arr[] = {1, 2, 2, 2, 2, 3, 4, 7, 8, 8};
    int n = sizeof(arr) / sizeof(arr[0]);
    int x = 2;
    printf("Number of occurrences of %d is %d\n", x, countOccurrences(arr, n, x));
    return 0;
}

        if (mid == 0 or x > arr[mid - 1]) and arr[mid] == x:

def binary_search_last(arr, left, right, x, n):

        elif x < arr[mid]:
            return binary_search_last(arr, left, mid - 1, x, n)
        else:

9. Median of two sorted arrays: There are 2 sorted arrays A and B; each of size n. Write an algorithm to find the median of the array obtained after merging the above 2 arrays (i.e. array of length 2n). The complexity should be O(log(n)).
================================================================================

--- C Implementation ---
#include <stdio.h>

#define MAX(a, b) ((a) > (b) ? (a) : (b))
#define MIN(a, b) ((a) < (b) ? (a) : (b))

double getMedian(int arp1[], int arp2[], int n) {
    int low = 0, high = n;
    while (low <= high) {
        int partition1 = (low + high) / 2;
        int partition2 = n - partition1;

        int maxLeft1 = (partition1 == 0) ? -1000000 : arp1[partition1 - 1];
        int minRight1 = (partition1 == n) ? 1000000 : arp1[partition1];

        int maxLeft2 = (partition2 == 0) ? -1000000 : arp2[partition2 - 1];
        int minRight2 = (partition2 == n) ? 1000000 : arp2[partition2];

        if (maxLeft1 <= minRight2 && maxLeft2 <= minRight1) {
            return (MAX(maxLeft1, maxLeft2) + MIN(minRight1, minRight2)) / 2.0;
        } else if (maxLeft1 > minRight2) {
            high = partition1 - 1;
        } else {
            low = partition1 + 1;
        }
    }
    return -1;
}

int main() {
    int ar1[] = {1, 12, 15, 26, 38};
    int ar2[] = {2, 13, 17, 30, 45};
    int n = sizeof(ar1) / sizeof(ar1[0]);
    printf("Median is %f\n", getMedian(ar1, ar2, n));
    return 0;
}


--- Python Implementation ---
def get_median(arr1, arr2, n):
    low = 0
    high = n
    
    while low <= high:
        partition1 = (low + high) // 2
        partition2 = n - partition1
        
        max_left1 = float('-inf') if partition1 == 0 else arr1[partition1 - 1]
        min_right1 = float('inf') if partition1 == n else arr1[partition1]
        
        max_left2 = float('-inf') if partition2 == 0 else arr2[partition2 - 1]
        min_right2 = float('inf') if partition2 == n else arr2[partition2]
        
        if max_left1 <= min_right2 and max_left2 <= min_right1:
            return (max(max_left1, max_left2) + min(min_right1, min_right2)) / 2.0
        elif max_left1 > min_right2:
            high = partition1 - 1
        else:
            low = partition1 + 1
            
    return -1

if __name__ == "__main__":
    ar1 = [1, 12, 15, 26, 38]
    ar2 = [2, 13, 17, 30, 45]
    n = len(ar1)
    print(f"Median is {get_median(ar1, ar2, n)}")

--- C Implementation ---
#include <stdio.h>
#include <string.h>

void sortDigits(int arr[], int n) {
    int count[10] = {0};
    int output[n];

    for (int i = 0; i < n; i++)
        count[arr[i]]++;

    for (int i = 1; i < 10; i++)
        count[i] += count[i - 1];

    for (int i = n - 1; i >= 0; i--) {
        output[count[arr[i]] - 1] = arr[i];
        count[arr[i]]--;
    }

    for (int i = 0; i < n; i++)
        arr[i] = output[i];
}

int main() {
    int arr[] = {3, 1, 4, 1, 5, 9, 2, 6, 5, 3};
    int n = sizeof(arr) / sizeof(arr[0]);
    sortDigits(arr, n);
    printf("Sorted digits: ");
    for(int i=0; i<n; i++) printf("%d ", arr[i]);
    printf("\n");
    return 0;
}


--- Python Implementation ---
def sort_digits(arr):
    count = [0] * 10
    output = [0] * len(arr)
    
    for digit in arr:
        count[digit] += 1
        
    for i in range(1, 10):
        count[i] += count[i - 1]
        
    for i in range(len(arr) - 1, -1, -1):
        output[count[arr[i]] - 1] = arr[i]
        count[arr[i]] -= 1
        
    for i in range(len(arr)):
        arr[i] = output[i]

11. Find neighbors of the median element in an array using the partitioning strategy of the Quick-Sorting method.
================================================================================

--- C Implementation ---
#include <stdio.h>

void swap(int* a, int* b) {
    int t = *a; *a = *b; *b = t;
}

int partition(int arr[], int l, int r) {
    int x = arr[r], i = l;
    for (int j = l; j <= r - 1; j++) {
        if (arr[j] <= x) {
            swap(&arr[i], &arr[j]);
            i++;
        }
    }
    swap(&arr[i], &arr[r]);
    return i;
}

int kthSmallest(int arr[], int l, int r, int k) {
    if (k > 0 && k <= r - l + 1) {
        int index = partition(arr, l, r);
        if (index - l == k - 1) return arr[index];
        if (index - l > k - 1) return kthSmallest(arr, l, index - 1, k);
        return kthSmallest(arr, index + 1, r, k - index + l - 1);
    }
    return -1;
}

int main() {
    int arr[] = {12, 3, 5, 7, 4, 19, 26};
    int n = sizeof(arr)/sizeof(arr[0]);
    int k = n / 2 + 1; // Median
    
    int median = kthSmallest(arr, 0, n - 1, k);
    
    int leftNeighbor = -10000;
    for (int i = 0; i < n / 2; i++) {
        if (arr[i] > leftNeighbor) leftNeighbor = arr[i];
    }
    
    int rightNeighbor = 10000;
    for (int i = n / 2 + 1; i < n; i++) {

def partition(arr, l, r):
    x = arr[r]
    i = l
    for j in range(l, r):
        if arr[j] <= x:
            arr[i], arr[j] = arr[j], arr[i]
            i += 1
    arr[i], arr[r] = arr[r], arr[i]
    return i

def kth_smallest(arr, l, r, k):
    if k > 0 and k <= r - l + 1:
        index = partition(arr, l, r)
        if index - l == k - 1:
            return arr[index]
        if index - l > k - 1:
            return kth_smallest(arr, l, index - 1, k)
        return kth_smallest(arr, index + 1, r, k - index + l - 1)
    return -1

    int i, w;
    int K[n + 1][W + 1];

    for (i = 0; i <= n; i++) {

    }
    return K[n][W];
}

void BellmanFord(struct Graph* graph, int src) {
    int V = graph->V;
    int E = graph->E;
    int dist[V];

    for (int i = 0; i < V; i++)
        dist[i] = INT_MAX;
    dist[src] = 0;

    for (int i = 1; i <= V - 1; i++) {
        for (int j = 0; j < E; j++) {
            int u = graph->edge[j].src;
            int v = graph->edge[j].dest;
            int weight = graph->edge[j].weight;
            if (dist[u] != INT_MAX && dist[u] + weight < dist[v])
                dist[v] = dist[u] + weight;
        }
    }

    for (int i = 0; i < E; i++) {
        int u = graph->edge[i].src;
        int v = graph->edge[i].dest;
        int weight = graph->edge[i].weight;
        if (dist[u] != INT_MAX && dist[u] + weight < dist[v]) {
            printf("Graph contains negative weight cycle\n");
            return;
        }
    }

    printf("Vertex Distance from Source\n");
    for (int i = 0; i < V; i++)
        printf("%d \t\t %d\n", i, dist[i]);
}

void addMat(int n, int** A, int** B, int** C) {
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            C[i][j] = A[i][j] + B[i][j];
}
void subMat(int n, int** A, int** B, int** C) {
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            C[i][j] = A[i][j] - B[i][j];
}

int** allocateMat(int n) {
    int** mat = (int**)malloc(n * sizeof(int*));
    for (int i = 0; i < n; i++)
        mat[i] = (int*)calloc(n, sizeof(int));
    return mat;
}

void strassen(int n, int** A, int** B, int** C) {
    if (n == 1) {
        C[0][0] = A[0][0] * B[0][0];
        return;
    }
    int k = n / 2;
    int **A11 = allocateMat(k), **A12 = allocateMat(k);
    int **A21 = allocateMat(k), **A22 = allocateMat(k);
    int **B11 = allocateMat(k), **B12 = allocateMat(k);
    int **B21 = allocateMat(k), **B22 = allocateMat(k);
    for (int i = 0; i < k; i++) {
        for (int j = 0; j < k; j++) {
            A11[i][j] = A[i][j]; A12[i][j] = A[i][j + k];
            A21[i][j] = A[i + k][j]; A22[i][j] = A[i + k][j + k];
            B11[i][j] = B[i][j]; B12[i][j] = B[i][j + k];
            B21[i][j] = B[i + k][j]; B22[i][j] = B[i + k][j + k];
        }
    }

    int **P1 = allocateMat(k), **P2 = allocateMat(k), **P3 = allocateMat(k);
    int **P4 = allocateMat(k), **P5 = allocateMat(k), **P6 = allocateMat(k), **P7 = allocateMat(k);
    int **T1 = allocateMat(k), **T2 = allocateMat(k);

    addMat(k, A11, A22, T1); addMat(k, B11, B22, T2); strassen(k, T1, T2, P1);
    addMat(k, A21, A22, T1); strassen(k, T1, B11, P2);
    subMat(k, B12, B22, T1); strassen(k, A11, T1, P3);
    subMat(k, B21, B11, T1); strassen(k, A22, T1, P4);
    addMat(k, A11, A12, T1); strassen(k, T1, B22, P5);
    subMat(k, A21, A11, T1); addMat(k, B11, B12, T2); strassen(k, T1, T2, P6);
    subMat(k, A12, A22, T1); addMat(k, B21, B22, T2); strassen(k, T1, T2, P7);

    for (int i = 0; i < k; i++) {
        for (int j = 0; j < k; j++) {
            C[i][j] = P1[i][j] + P4[i][j] - P5[i][j] + P7[i][j];
            C[i][j + k] = P3[i][j] + P5[i][j];
            C[i + k][j] = P2[i][j] + P4[i][j];
            C[i + k][j + k] = P1[i][j] - P2[i][j] + P3[i][j] + P6[i][j];
        }
    }
}

int nextPowerOf2(int n) {
    int p = 1;
    while (p < n) p *= 2;
    return p;
}

    A12 = [row[k:] for row in A[:k]]
    A21 = [row[:k] for row in A[k:]]
    A22 = [row[k:] for row in A[k:]]

    B11 = [row[:k] for row in B[:k]]
    B12 = [row[k:] for row in B[:k]]
    B21 = [row[:k] for row in B[k:]]
    B22 = [row[k:] for row in B[k:]]

    def add(M1, M2):
        return [[M1[i][j] + M2[i][j] for j in range(len(M1))] for i in range(len(M1))]

        return [[M1[i][j] - M2[i][j] for j in range(len(M1))] for i in range(len(M1))]

bool isSubsetSum(int set[], int n, int sum) {
    bool subset[n + 1][sum + 1];

    for (int i = 0; i <= n; i++)
        subset[i][0] = true;
    for (int i = 1; i <= sum; i++)
        subset[0][i] = false;

    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= sum; j++) {
            if (j < set[i - 1])
                subset[i][j] = subset[i - 1][j];
            else
                subset[i][j] = subset[i - 1][j] || subset[i - 1][j - set[i - 1]];
        }
    }
    return subset[n][sum];
}

--- C Implementation ---
#include <stdio.h>
#include <limits.h>

#define MAX 9999
#define n 4

int dist[n][n] = {
    {0, 10, 15, 20},
    {10, 0, 35, 25},
    {15, 35, 0, 30},
    {20, 25, 30, 0}
};

int dp[16][4];
int VISITED_ALL = (1 << n) - 1;

int tsp(int mask, int pos) {
    if (mask == VISITED_ALL) {
        return dist[pos][0];
    }
    if (dp[mask][pos] != -1) {
        return dp[mask][pos];
    }

    int ans = MAX;
    for (int city = 0; city < n; city++) {
        if ((mask & (1 << city)) == 0) {
            int newAns = dist[pos][city] + tsp(mask | (1 << city), city);
            if (newAns < ans) ans = newAns;
        }
    }
    return dp[mask][pos] = ans;
}

int main() {
    for (int i = 0; i < (1 << n); i++) {
        for (int j = 0; j < n; j++) {
            dp[i][j] = -1;
        }
    }
    printf("Minimum weight Hamiltonian Cycle (TSP) is: %d\n", tsp(1, 0));
    return 0;
}


--- Python Implementation ---
import sys

dist = [
    [0, 10, 15, 20],
    [10, 0, 35, 25],
    [15, 35, 0, 30],
    [20, 25, 30, 0]
]

n = 4
VISITED_ALL = (1 << n) - 1
dp = [[-1] * n for _ in range(1 << n)]

def tsp(mask, pos):
    if mask == VISITED_ALL:
        return dist[pos][0]
    
    if dp[mask][pos] != -1:
        return dp[mask][pos]
        
    ans = sys.maxsize
    for city in range(n):
        if (mask & (1 << city)) == 0:
            new_ans = dist[pos][city] + tsp(mask | (1 << city), city)
            ans = min(ans, new_ans)
            
    dp[mask][pos] = ans
    return ans

void floydWarshall(int dist[][V]) {
    for (int k = 0; k < V; k++) {
        for (int i = 0; i < V; i++) {
            for (int j = 0; j < V; j++) {
                if (dist[i][k] + dist[k][j] < dist[i][j])
                    dist[i][j] = dist[i][k] + dist[k][j];
            }
        }
    }
    printSolution(dist);
}

#define MAX(a, b) ((a) > (b) ? (a) : (b))

int lcs(char *X, char *Y, int m, int n) {
    int L[m + 1][n + 1];

    for (int i = 0; i <= m; i++) {
        for (int j = 0; j <= n; j++) {
            if (i == 0 || j == 0)
                L[i][j] = 0;
            else if (X[i - 1] == Y[j - 1])
                L[i][j] = L[i - 1][j - 1] + 1;
            else
                L[i][j] = MAX(L[i - 1][j], L[i][j - 1]);
        }
    }
    return L[m][n];
}

#include <stdio.h>
#include <string.h>

void computeLPSArray(char* pat, int M, int* lps) {
    int len = 0;
    lps[0] = 0;
    int i = 1;

    while (i < M) {
        if (pat[i] == pat[len]) {
            len++;
            lps[i] = len;
            i++;
        } else {
            if (len != 0) {
                len = lps[len - 1];
            } else {
                lps[i] = 0;
                i++;
            }
        }
    }
}

void KMPSearch(char* pat, char* txt) {
    int M = strlen(pat);
    int N = strlen(txt);
    int lps[M];

    computeLPSArray(pat, M, lps);

    int i = 0, j = 0;
    while (i < N) {
        if (pat[j] == txt[i]) {
            j++;
            i++;
        }
        if (j == M) {
            printf("Found pattern at index %d\n", i - j);
            j = lps[j - 1];
        } else if (i < N && pat[j] != txt[i]) {
            if (j != 0)
                j = lps[j - 1];
            else
                i = i + 1;
        }
    }
}
