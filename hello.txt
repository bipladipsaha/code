import java.util.Scanner;

class Utilities {
    // Compute a mathematical result using iterative logic
    // demonstrate modularity by implementing the logic using methods and separate classes.
    public long calculateFactorial(int n) {
        long fact = 1;
        for (int i = 1; i <= n; i++) {
            fact *= i;
        }
        return fact;
    }
}

public class UtilitySystem {
    public static void main(String[] args) {
        // Display an appropriate welcome message at the start of execution.
        System.out.println("**************************************************");
        System.out.println("* Welcome to the Educational Institute Utility System *");
        System.out.println("**************************************************\n");

        // Perform basic arithmetic processing using declared variables.
        int declaredA = 50;
        int declaredB = 25;
        System.out.println("--- Arithmetic Processing ---");
        System.out.println("Addition: " + declaredA + " + " + declaredB + " = " + (declaredA + declaredB));
        System.out.println("Subtraction: " + declaredA + " - " + declaredB + " = " + (declaredA - declaredB));
        
        // Demonstrate modularity
        Utilities utils = new Utilities();
        int factTarget = 5;
        System.out.println("\n--- Modular Factorial Calculation ---");
        System.out.println("Factorial of " + factTarget + " is " + utils.calculateFactorial(factTarget));
        
        // Accept user input via command-line arguments.
        System.out.println("\n--- Command Line Inputs ---");
        if (args.length > 0) {
            System.out.println("Received " + args.length + " command line arguments.");
            for (String arg : args) {
                System.out.println("Arg: " + arg);
            }
        } else {
            System.out.println("No command line arguments passed.");
        }

        // Accept user input both through the keyboard
        System.out.println("\n--- Keyboard Input & Classification ---");
        Scanner scanner = new Scanner(System.in);
        System.out.print("Enter a numeric value to classify: ");
        
        if (scanner.hasNextInt()) {
            int inputNum = scanner.nextInt();
            // Classify numeric input based on given conditions.
            if (inputNum > 0) {
                System.out.println("Classification: The number is Positive.");
                if (inputNum % 2 == 0) System.out.println("It is also an Even number.");
                else System.out.println("It is also an Odd number.");
            } else if (inputNum < 0) {
                System.out.println("Classification: The number is Negative.");
            } else {
                System.out.println("Classification: The number is Zero.");
            }
        } else {
            System.out.println("Invalid numeric input.");
        }
        
        scanner.close();
    }

class TestObject {
    int data = 50;
    
    // Simulating call by reference by passing an object reference
    void changeReference(TestObject obj) {
        obj.data = obj.data + 100;
    }
}

public class CallByValueReference {
    // Call by value method
    static void changeValue(int data) {
        data = data + 100;
        System.out.println("Inside changeValue method: " + data);
    }
    
    public static void main(String[] args) {
        // Call by value demonstration
        int primitiveData = 50;
        System.out.println("Before call by value: " + primitiveData);
        changeValue(primitiveData);
        System.out.println("After call by value (primitive value unchanged): " + primitiveData);
        
        System.out.println("-------------------------");
        
        // Call by reference demonstration
        TestObject obj = new TestObject();
        System.out.println("Before call by reference: " + obj.data);
        obj.changeReference(obj);
        System.out.println("After call by reference (object state changed): " + obj.data);
    }
}

public class MatrixMultiplication {
    public static void main(String[] args) {
        int[][] a = {{1, 2, 3}, {4, 5, 6}}; // 2x3 matrix
        int[][] b = {{7, 8}, {9, 10}, {11, 12}}; // 3x2 matrix
        
        int rowsA = a.length;
        int colsA = a[0].length;
        int colsB = b[0].length;
        
        int[][] result = new int[rowsA][colsB];
        
        System.out.println("Product of matrices:");
        for (int i = 0; i < rowsA; i++) {
            for (int j = 0; j < colsB; j++) {
                result[i][j] = 0;
                for (int k = 0; k < colsA; k++) {
                    result[i][j] += a[i][k] * b[k][j];
                }
                System.out.print(result[i][j] + " ");
            }
            System.out.println();
        }
    }
}

import java.util.Scanner;

public class StudentMarksSystem {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.print("Enter the number of students: ");
        int n = scanner.nextInt();
        
        if (n <= 0) {
            System.out.println("Invalid number of students.");
            return;
        }
        
        int[] marks = new int[n];
        int total = 0;
        int highest = Integer.MIN_VALUE;
        int lowest = Integer.MAX_VALUE;
        
        // Dynamic Input & Validation
        for (int i = 0; i < n; i++) {
            System.out.print("Enter marks for student " + (i + 1) + " (0-100): ");
            int m = scanner.nextInt();
            
            while (m < 0 || m > 100) {
                System.out.print("Invalid marks. Please enter a value between 0 and 100: ");
                m = scanner.nextInt();
            }
            
            marks[i] = m;
            total += m;
            
            if (m > highest) highest = m;
            if (m < lowest) lowest = m;
        }
        
        double average = (double) total / n;
        
        System.out.println("\n===== Statistical Information =====");
        System.out.println("Total Marks obtained: " + total);
        System.out.println("Average Marks: " + average);
        System.out.println("Highest Marks: " + highest);
        System.out.println("Lowest Marks: " + lowest);
        
        System.out.println("\n===== Detailed Student Records =====");
        for (int i = 0; i < n; i++) {
            String grade;
            if (marks[i] >= 90) grade = "A";
            else if (marks[i] >= 80) grade = "B";
            else if (marks[i] >= 70) grade = "C";
            else if (marks[i] >= 60) grade = "D";
            else grade = "F (Fail)";
            
            System.out.println("Student " + (i + 1) + "  |  Marks: " + marks[i] + "  |  Grade: " + grade);
        }
        
        scanner.close();
    }
}

    public static void main(String[] args) {
        int num = 153;
        int original = num;
        int sum = 0;
        
        int digits = String.valueOf(num).length();
        
        while (num != 0) {
            int digit = num % 10;

        if (sum == original) {

        } else {
            System.out.println(original + " is not an Armstrong Number.");
        }
    }
}

import java.util.Scanner;

public class DynamicTextEditor {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter initial content for the text editor: ");
        String initialContent = sc.nextLine();
        
        StringBuffer editor = new StringBuffer(initialContent);
        
        // Display current text, length, and capacity
        System.out.println("\nCurrent Text: " + editor);
        System.out.println("Length: " + editor.length());
        System.out.println("Capacity: " + editor.capacity());
        
        // Add additional text
        System.out.print("\nEnter text to append: ");
        String appendText = sc.nextLine();
        editor.append(appendText);
        System.out.println("Updated Text: " + editor);
        
        // Access specific character
        System.out.print("\nEnter index to get character: ");
        int index = sc.nextInt();
        if (index >= 0 && index < editor.length()) {
            System.out.println("Character at index " + index + ": " + editor.charAt(index));
        } else {
            System.out.println("Invalid index.");
        }
        
        // Insert new text
        System.out.print("\nEnter text to insert: ");
        sc.nextLine(); // clear buffer
        String insertText = sc.nextLine();
        System.out.print("Enter position to insert: ");
        int insertPos = sc.nextInt();
        if (insertPos >= 0 && insertPos <= editor.length()) {
            editor.insert(insertPos, insertText);
            System.out.println("Updated Text: " + editor);
        } else {
            System.out.println("Invalid position.");
        }
        
        // Search for a word/character sequence
        System.out.print("\nEnter sequence to search: ");
        sc.nextLine(); // clear buffer
        String searchSeq = sc.nextLine();
        int pos = editor.indexOf(searchSeq);
        if (pos != -1) {
            System.out.println("Sequence found at position: " + pos);
        } else {
            System.out.println("Sequence not found.");
        }
        
        sc.close();
    }
}

        int num = 29;

        if (num <= 1) {
            isPrime = false;
        } else {
            for (int i = 2; i <= Math.sqrt(num); i++) {

                    break;

        }
        
        if (isPrime) {
            System.out.println(num + " is a Prime number.");
        } else {
            System.out.println(num + " is not a Prime number.");

        System.out.println("Empty Document Created. Initial Capacity: " + content.capacity());
    }
    
    Document(String initialText) {
        content = new StringBuffer(initialText);
        System.out.println("Document Created. Initial Content Length: " + content.length());
    }
    
    // Method overloading for adding content
    void addContent(String text) {
        content.append(text);
        System.out.println("Text Added (String). Current length: " + content.length());
    }
    
    void addContent(char c) {
        content.append(c);
        System.out.println("Text Added (Char). Current length: " + content.length());
    }
    
    void insertContent(int index, String text) {
        if (index >= 0 && index <= content.length()) {
            content.insert(index, text);
            System.out.println("Text inserted at " + index);
        } else {
            System.out.println("Invalid index!");
        }
    }
    
    void searchAndReplace(String search, String replace) {
        int index = content.indexOf(search);
        if (index != -1) {
            content.replace(index, index + search.length(), replace);
            System.out.println("Replaced '" + search + "' with '" + replace + "'");
        } else {
            System.out.println("Text not found!");
        }
    }
    
    Document extract(int start, int end) {
        if (start >= 0 && end <= content.length() && start <= end) {
            String extracted = content.substring(start, end);
            return new Document(extracted); // Returning current class instance type
        }
        System.out.println("Invalid range for extraction.");
        return null;
    }
    
    String generateFinalDocument() {
        return content.toString();
    }
}

public class DocumentProcessingSystem {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter initial document content: ");
        String initText = sc.nextLine();
        Document doc = new Document(initText);
        
        doc.addContent(" -- Added line -- ");
        doc.addContent('Z');
        
        System.out.println("\nFinal Processed Document String:");
        System.out.println(doc.generateFinalDocument());
        
        System.out.println("\nExtracted Document (0 to 5):");
        Document extracted = doc.extract(0, 5);
        if (extracted != null) {
            System.out.println(extracted.generateFinalDocument());
        }
        sc.close();
    }
}

import java.util.Scanner;

class ThreeDObject {
    public double wholeSurfaceArea() { return 0.0; }
    public double volume() { return 0.0; }
}

class Box extends ThreeDObject {
    double l, w, h;
    Box(double l, double w, double h) {
        this.l = l; this.w = w; this.h = h;
    }
    public double wholeSurfaceArea() { return 2 * (l * w + w * h + h * l); }
    public double volume() { return l * w * h; }
}

class Cube extends ThreeDObject {
    double side;
    Cube(double side) { this.side = side; }
    public double wholeSurfaceArea() { return 6 * side * side; }
    public double volume() { return side * side * side; }
}

class Cylinder extends ThreeDObject {
    double r, h;
    Cylinder(double r, double h) {
        this.r = r; this.h = h;
    }
    public double wholeSurfaceArea() { return 2 * Math.PI * r * (r + h); }
    public double volume() { return Math.PI * r * r * h; }
}

class Cone extends ThreeDObject {
    double r, h;
    Cone(double r, double h) {
        this.r = r; this.h = h;
    }
    public double wholeSurfaceArea() {
        double slantHeight = Math.sqrt(r * r + h * h);
        return Math.PI * r * (r + slantHeight);
    }
    public double volume() { return (1.0/3.0) * Math.PI * r * r * h; }
}

 * Java doesn't support multiple inheritance with classes to avoid the Diamond Problem.
 * If class C extends both class A and class B, and both A and B have a method with the same signature,
 * the compiler wouldn't know which method to inherit. 
 * Therefore, Java uses Interfaces to achieve multiple inheritance.

    void print();
}

interface Showable {
    void show();
}

class A implements Printable, Showable {
    public void print() {
        System.out.println("Hello from Printable");
    }
    public void show() {
        System.out.println("World from Showable");
    }
}

public class MultipleInheritanceJustification {
    public static void main(String[] args) {
        System.out.println("Multiple inheritance achieved through interfaces.");
        A obj = new A();
        obj.print();
        obj.show();
    }
}

public class LongestPalindromicSubstring {
    public static String longestPalindrome(String s) {
        if (s == null || s.length() < 1) return "";
        int start = 0, end = 0;
        for (int i = 0; i < s.length(); i++) {
            int len1 = expandAroundCenter(s, i, i);
            int len2 = expandAroundCenter(s, i, i + 1);
            int len = Math.max(len1, len2);
            if (len > end - start) {
                start = i - (len - 1) / 2;
                end = i + len / 2;
            }
        }
        return s.substring(start, end + 1);
    }
    
    private static int expandAroundCenter(String s, int left, int right) {
        int L = left, R = right;
        while (L >= 0 && R < s.length() && s.charAt(L) == s.charAt(R)) {
            L--;
            R++;
        }
        return R - L - 1;
    }

    public static void main(String[] args) {
        String testStr = "babad";
        System.out.println("Input String: " + testStr);
        System.out.println("Longest Palindromic Substring: " + longestPalindrome(testStr));
    }
}

import java.util.HashSet;
import java.util.Set;

public class LongestSubstring {
    public static int lengthOfLongestSubstring(String s) {
        Set<Character> set = new HashSet<>();
        int max = 0, left = 0;
        
        for (int right = 0; right < s.length(); right++) {
            while (set.contains(s.charAt(right))) {
                set.remove(s.charAt(left));
                left++;
            }
            set.add(s.charAt(right));
            max = Math.max(max, right - left + 1);
        }
        
        return max;
    }
    
    public static void main(String[] args) {
        String testInput = "abcabcbb";
        System.out.println("Input: " + testInput);
        System.out.println("Length of longest substring without repeating characters: " + lengthOfLongestSubstring(testInput));
    }
}

import java.util.ArrayList;

class Book {
    int id;
    String title;
    boolean isAvailable = true;
    
    Book(int id, String title) {
        this.id = id;
        this.title = title;
    }
}

class Member {
    int memberId;
    String name;
    
    Member(int id, String name) {
        this.memberId = id;
        this.name = name;
    }
}

class Library {
    List<Book> books = new ArrayList<>();
    
    void addBook(Book b) {
        books.add(b);
    }
    
    // Synchronized borrow
    synchronized void borrowBook(int bookId, String memberName) {
        for (Book b : books) {
            if (b.id == bookId) {
                if (b.isAvailable) {
                    System.out.println(memberName + " borrowed book: " + b.title);
                    b.isAvailable = false;
                } else {
                    System.out.println("Sorry " + memberName + ", book '" + b.title + "' is currently unavailable.");
                }
                return;
            }
        }
        System.out.println("Book not found.");
    }
    
    // Synchronized return
    synchronized void returnBook(int bookId, String memberName) {
        for (Book b : books) {
            if (b.id == bookId) {
                b.isAvailable = true;
                System.out.println(memberName + " returned book: " + b.title);
                return;
            }
        }
    }
}

class MemberThread extends Thread {
    Library lib;
    int bookId;
    String memberName;
    boolean isBorrowing;

    MemberThread(Library lib, int bookId, String memberName, boolean isBorrowing) {
        this.lib = lib;
        this.bookId = bookId;
        this.memberName = memberName;
        this.isBorrowing = isBorrowing;
    }

    public void run() {
        if (isBorrowing) {
            lib.borrowBook(bookId, memberName);
        } else {
            lib.returnBook(bookId, memberName);
        }
    }
}

    public static void main(String[] args) {
        Library lib = new Library();
        lib.addBook(new Book(1, "Deep Learning"));
        lib.addBook(new Book(2, "Java Concurrency"));
        
        MemberThread t1 = new MemberThread(lib, 1, "Alice", true);
        MemberThread t2 = new MemberThread(lib, 1, "Bob", true); // Will fail because Alice took it
        MemberThread t3 = new MemberThread(lib, 1, "Alice", false); // Alice returns it
        
        try {
            t1.start();
            t1.join();
            t2.start();
            t2.join();
            t3.start();
            t3.join();
            
            // Now Bob tries again
            MemberThread t4 = new MemberThread(lib, 1, "Bob", true);
            t4.start();
            
        } catch (InterruptedException e) {

        }
    }
}

class Printer {
    // Synchronized block inside a method
    void printDocs(int copies, String docName) {
        synchronized(this) { // Only one thread at a time can execute this block
            for (int i = 1; i <= copies; i++) {
                try {
                    Thread.sleep(100);
                } catch (Exception e) {
                    System.out.println(e);
                }
                System.out.println("Printing " + docName + " copy #" + i);
            }
        }
    }
}

            t1.join(); // Wait for t1 to finish

    }
}

class PriorityCounter extends Thread {
    volatile boolean running = true;
    long count = 0;
    
    public void run() {
        while (running) {
            count++;
        }
    }
    
    public void stopThread() {
        running = false;
    }
}

public class ThreadPriorityCounter {
    public static void main(String[] args) {
        PriorityCounter t1 = new PriorityCounter();
        PriorityCounter t2 = new PriorityCounter();
        PriorityCounter t3 = new PriorityCounter();
        PriorityCounter t4 = new PriorityCounter();
        
        t1.setPriority(1);
        t2.setPriority(3);
        t3.setPriority(5);
        t4.setPriority(7);
        
        t1.start();
        t2.start();
        t3.start();
        t4.start();
        
        try {
            // Update counter in each thread for 10 ms
            Thread.sleep(10);
        } catch (InterruptedException e) {
            System.out.println(e);
        }
        
        t1.stopThread();
        t2.stopThread();
        t3.stopThread();
        t4.stopThread();
        
        System.out.println("Thread 1 (Priority 1) Count: " + t1.count);
        System.out.println("Thread 2 (Priority 3) Count: " + t2.count);
        System.out.println("Thread 3 (Priority 5) Count: " + t3.count);
        System.out.println("Thread 4 (Priority 7) Count: " + t4.count);
    }
}

    }
